module digital_clock (
    input clk,
    input rst_n,
    input tick_1s,
    output reg [5:0] seconds,
    output reg [5:0] minutes,
    output reg [4:0] hours
);

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            seconds <= 0;
        else if (tick_1s) begin
            if (seconds == 59)
                seconds <= 0;
            else
                seconds <= seconds + 1;
        end
    end

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            minutes <= 0;
        else if (tick_1s && seconds == 59) begin
            if (minutes == 59)
                minutes <= 0;
            else
                minutes <= minutes + 1;
        end
    end

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            hours <= 0;
        else if (tick_1s && seconds == 59 && minutes == 59) begin
            if (hours == 23)
                hours <= 0;
            else
                hours <= hours + 1;
        end
    end

endmodule
